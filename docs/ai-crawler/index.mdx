## üîß Technical Details

### **Crawlee.dev Integration**
```javascript
// Our crawler is built with Crawlee.dev
const crawler = new PlaywrightCrawler({
  async requestHandler({ page, sendRequest }) {
    // Intelligent content extraction
    const data = await page.evaluate(() => {
      return {
        title: document.title,
        content: document.body.textContent,
        links: Array.from(document.links).map(link => link.href),
        images: Array.from(document.images).map(img => img.src)
      };
    });
    
    // Send to MCP server for AI analysis
    await sendRequest({
      url: 'mcp://crawler/analyze',
      method: 'POST',
      body: data
    });
  }
});
```

### **MCP Server Integration**
```typescript
interface CrawlerMCPServer {
  // Available endpoints
  endpoints: {
    '/crawl/github': (repo: string) => Promise<RepoData>;
    '/crawl/documentation': (url: string) => Promise<DocData>;
    '/crawl/website': (url: string) => Promise<WebData>;
    '/analyze/trends': (data: any[]) => Promise<TrendAnalysis>;
    '/export/results': (format: 'json' | 'csv') => Promise<ExportData>;
  }
}
```
